# Commands to clean up your Git repository
# Run these commands in your terminal one by one:

# 1. Remove node_modules from Git tracking (but keep locally)
git rm -r --cached node_modules

# 2. Remove .env file from Git tracking (but keep locally)
git rm --cached .env

# 3. Remove any other unwanted files that were accidentally committed
# git rm --cached filename.ext

# 4. Add the changes to staging
git add .

# 5. Commit the cleanup
git commit -m "Remove node_modules, .env and other files from repository"

# 6. Push the changes
git push origin main

# Note: After this, your repository will be clean and future commits
# will respect the .gitignore file properly.
